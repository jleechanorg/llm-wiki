## Testing
- ARGUMENTS="" bash -lc '(
    set -euo pipefail

    ARGUMENTS_VALUE="${ARGUMENTS-}"

    candidates=(
        "$(pwd -P)/.claude/commands/gstatus.py"
    )

    if git_root=$(git rev-parse --show-toplevel 2>/dev/null); then
        candidates+=("$git_root/.claude/commands/gstatus.py")
    fi

    if [ -n "${CLAUDE_REPO_ROOT:-}" ]; then
        candidates+=("${CLAUDE_REPO_ROOT%/}/.claude/commands/gstatus.py")
    fi

    candidates+=("$HOME/.claude/commands/gstatus.py")

    for candidate in "${candidates[@]}"; do
        if [ -f "$candidate" ]; then
            python3 "$candidate" "$ARGUMENTS_VALUE"
            exit $?
        fi
    done

    echo "Error: Unable to locate gstatus.py. Checked:" >&2
    for candidate in "${candidates[@]}"; do
        echo "  - $candidate" >&2
    done
    exit 1
)'
