def update(self, data):
    for key, value in data.items():
        if '.' in key:
            # Handle 'settings.gemini_model' syntax
            parts = key.split('.')
            current = self._data
            for part in parts[:-1]:
                if part not in current:
                    current[part] = {}
                current = current[part]
            current[parts[-1]] = value
