+        # XP/Gold fields have min default of 0

+        self.assertEqual(DNC.convert_value("gold", "500gp"), 500)
+        self.assertEqual(DNC.convert_value("xp_current", "850XP"), 850)
+        self.assertEqual(DNC.convert_value("xp_to_next_level", "300 pts"), 300)
+
+    def test_robust_ac_status_extraction(self):
+        self.assertEqual(DNC.convert_value("ac", "15 AC"), 15)
+        self.assertEqual(DNC.convert_value("armor_class", "No Armor (10)"), 10)

+    def test_robust_fallback_behavior(self):
+        # Fields with no numeric content should return the default

+        self.assertEqual(DNC.convert_value("gold", "None"), 0)
+        self.assertEqual(DNC.convert_value("xp", None), 0)

+            "loot": ["item1", "item2"] # Non-numeric fields preserved

+#!/usr/bin/env python3
+"""
+Rewards Box Robustness E2E Test
+
+Validates that when the LLM outputs hallucinated or messy reward fields
+(e.g., 'xpAmount': '1,000 XP'), the unified defensive numeric converter 
+safely normalizes them so they render visibly at the UI output layer.
+"""
+
+import sys
+from typing import Any
+from pathlib import Path
+
+# Standard root-level testing_mcp path bootstrap
+sys.path.insert(0, str(Path(__file__).parent.parent))
+
+from testing_mcp.lib.base_test import MCPTestBase, TestContext, run_test
+from mvp_site.world_logic import normalize_rewards_box_for_ui
+
+class RewardsBoxRobustnessTest(MCPTestBase):
+    TEST_NAME = "rewards_box_robustness_real_e2e"
+    MODEL = "gemini-3-flash-preview"
+    DESCRIPTION = "Tests central robust numeric conversion on hallucinated LLM payloads"
+
+    def run_scenarios(self, ctx: TestContext) -> list[dict[str, Any]]:
+        results = []
+        # Run the internal unit tests to ensure all edge cases are strictly validated
+        results.append(self.run_internal_unit_tests())
+        
+        # Run a live E2E test attempting to provoke a messy LLM response
+        results.append(self.run_e2e_hallucination_provocation(ctx))
+        return results
+
+    def run_internal_unit_tests(self) -> dict[str, Any]:
+        """Runs the original unit-test checks to strictly validate all theoretical permutations."""
+        print("\n=== SCENARIO: internal_unit_tests ===")
+        errors = []
+        
+        # We test combinations here so we don't need to try and force the LLM to hallucinate every single edge case
+        tests = [
+            ("messy_strings", {"xp_gained": "1,000 XP", "gold": "250 GP", "loot": ["Shield"]}, 1000, 250),
+            ("fallback_xp_gp", {"xp": 500, "gp": 100}, 500, 100),
+            ("fallback_aliases", {"xpAmount": "1,250 XP", "gold_awarded": "300"}, 1250, 300),
+            ("empty_primary_fallback", {"xp_gained": "", "xp": 42}, 42, 0),
+            ("nan_primary_fallback", {"xp_gained": float("nan"), "xp": 77}, 77, 0)
+        ]
+
+        for name, payload, expected_xp, expected_gold in tests:
+            try:
+                norm = normalize_rewards_box_for_ui(payload)
+                if not norm:
+                    errors.append(f"Unit Test {name} failed: Returned None")
+                elif norm["xp_gained"] != expected_xp or norm["gold"] != expected_gold:
+                    errors.append(f"Unit Test {name} failed: Expected XP {expected_xp}, Gold {expected_gold}, got {norm}")
+            except Exception as e:
+                errors.append(f"Unit Test {name} raised exception: {e}")
+
+        passed = len(errors) == 0
+        return {
+            "name": "internal_unit_tests",
+            "campaign_id": "none",
+            "passed": passed,
+            "errors": errors
+        }
+
